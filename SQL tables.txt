CREATE TABLE Patients(
id INT NOT NULL PRIMARY KEY,
Gender VARCHAR(255),
YearOfBirth YEAR,
Smoker VARCHAR(255),
DateOfDiagnosis DATE,
DiagnosisSummary VARCHAR(255),
Diagnosis VARCHAR(255),
AgeAtFirstVisit TINYINT,
Dataset VARCHAR(255),
TypeOfRecruit VARCHAR(255)
);

CREATE TABLE Visit(
Mobility INT(1),
SelfCare INT(1),
PainDiscourse INT(1),
UsualActivity INT(1),
AnxietyDepression INT(1),
DateOfVisit DATE,
EQ5D VARCHAR(255),
HealthScore INT(5),
FirstVisit DATE,
Occupation VARCHAR(255),
Cyclosphamide VARCHAR(255),
Steroids VARCHAR(255),
ImmunoSuppressent VARCHAR(255),
Prophlaxis VARCHAR(255),
FormType VARCHAR(255),
DiseaseActivity VARCHAR(255),
CRP VARCHAR(255),
BVAS VARCHAR(255),
Creatinine VARCHAR(255),
eGFR VARCHAR(255),
ID INT NOT NULL,
HealthIndex DECIMAL(6,3)
);

ALTER TABLE visit
ADD FOREIGN KEY ID(ID)
REFERENCES Patients(id);

ALTER TABLE visit
ADD PRIMARY KEY(DateOfVisit, ID);
